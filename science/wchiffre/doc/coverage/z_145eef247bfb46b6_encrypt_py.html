1""" 

2Fenster-Chiffre – Verschluesselung (Algorithmus 1, Definition 3.1). 

3 

4Enc(pk, s, M) -> P 

5 

6Ablauf: 

7 1. Zerlege M in Bloecke M_1,...,M_r (jeder Block < n, bitweises Framing) 

8 2. Berechne Gesamtlaenge l = minimale Paketanzahl fuer genau r echte Slots 

9 3. Fuer i in 1..l: 

10 falls (i-1+s) mod k == 0 -> Dummy-Paket (Zufallszahl in [0,n)) 

11 sonst -> C_j = M_j^e mod n 

12 

13Blockzerlegung (message_to_blocks): 

14 Jeder Block traegt B = floor(log2(n)) Nutzbits. 

15 Framing: [16-Bit-Laengenfeld | Nachrichtenbits | 0-Fuellbits] 

16 Das Laengenfeld kodiert len(message) in Bytes (max 65535). 

17 Damit ist die Rekonstruktion eindeutig fuer beliebig kleine n (auch n=55). 

18"""

32def _dummy_int(n: int) -> int: 

33 """Erzeugt eine gleichverteilte Zufallszahl in [0, n).""" 

34 byte_len = math.ceil(n.bit_length() / 8) 

35 raw = os.urandom(byte_len) 

36 return int.from_bytes(raw, byteorder="big") % n

52def _compute_packet_count(r: int, k: int, s: int) -> int: 

53 """Minimale Paketanzahl l so dass genau r echte Slots entstehen. 

54 

55 Echte Slots = Positionen i in [1..l] mit (i-1+s) % k != 0. 

56 Die Approximation l = ceil(r*k/(k-1)) ist nur fuer s=0 exakt; 

57 diese Funktion korrigiert fuer alle Offsets. 

58 

59 Parameters 

60 ---------- 

61 r : int Anzahl der echten Bloecke (>= 0). 

62 k : int Fenster-Abstand (>= 2). 

63 s : int Offset (0 <= s < k). 

64 

65 Returns 

66 ------- 

67 int Minimales l >= 0 mit exakt r echten Slots. 

68 """ 

69 if r == 0: 

70 return 0 

71 l = math.ceil(r * k / (k - 1)) 

72 while True: 

73 real = sum(1 for i in range(1, l + 1) if (i - 1 + s) % k != 0) 

74 if real == r: 

75 return l 

76 if real < r: 

77 l += 1 

78 else: 

79 real_prev = sum(1 for i in range(1, l) if (i - 1 + s) % k != 0) 

80 return l - 1 if real_prev == r else l

87def message_to_blocks(message: bytes, n: int) -> List[int]: 

88 """Zerlegt eine Bytefolge in Klartextbloecke < n. 

89 

90 Jeder Block traegt B = floor(log2(n)) Nutzbits. 

91 Framing: [16-Bit-Laengenfeld | Nachrichtenbits | 0-Fuellbits auf Vielfaches von B]. 

92 

93 Parameters 

94 ---------- 

95 message : bytes Klartextnachricht (max 65535 Bytes). 

96 n : int RSA-Modulus (>= 3). 

97 

98 Returns 

99 ------- 

100 List[int] Blockliste, jeder Wert in [0, 2^B - 1] < n. 

101 """ 

102 if n < 3: 

103 raise ValueError(f"RSA-Modulus n muss >= 3 sein, erhalten: {n}") 

104 

105 B = _block_bits(n) 

106 if B < 1: 

107 raise ValueError(f"n={n} zu klein fuer Blockzerlegung.") 

108 

109 msg_len = len(message) 

110 if msg_len > 65535: 

111 raise ValueError(f"Nachricht zu lang: {msg_len} Bytes (max 65535).") 

112 

113 # Bitstream: 16-Bit-Laengenfeld || Nachrichtenbits 

114 msg_int = int.from_bytes(message, byteorder="big") if message else 0 

115 payload = (msg_len << (msg_len * 8)) | msg_int 

116 payload_bit_len = 16 + msg_len * 8 

117 

118 # Rechts mit 0-Bits auf Vielfaches von B auffuellen 

119 remainder = payload_bit_len % B 

120 if remainder != 0: 

121 pad = B - remainder 

122 payload <<= pad 

123 payload_bit_len += pad 

124 

125 num_blocks = payload_bit_len // B 

126 mask = (1 << B) - 1 

127 

128 blocks: List[int] = [] 

129 for i in range(num_blocks - 1, -1, -1): 

130 val = (payload >> (i * B)) & mask 

131 assert val < n, f"Interner Fehler: Blockwert {val} >= n={n}" 

132 blocks.append(val) 

133 

134 return blocks

137def blocks_to_message(blocks: List[int], n: int) -> bytes: 

138 """Rekonstruiert eine Bytefolge aus Klartextbloecken (Inverse von message_to_blocks). 

139 

140 Parameters 

141 ---------- 

142 blocks : List[int] Entschluesselten Bloecke. 

143 n : int RSA-Modulus. 

144 

145 Returns 

146 ------- 

147 bytes Urspruengliche Nachricht. 

148 

149 Raises 

150 ------ 

151 ValueError Bei ungueltiger Blockstruktur. 

152 """ 

153 if not blocks: 

154 raise ValueError("Ungueltiges Padding: 0x80-Byte nicht gefunden.") 

155 

156 B = _block_bits(n) 

157 total_bits = len(blocks) * B 

158 

159 if total_bits < 16: 

160 raise ValueError("Ungueltiges Padding: 0x80-Byte nicht gefunden.") 

161 

162 # Alle Bloecke zu einem Integer zusammensetzen 

163 combined = 0 

164 for b in blocks: 

165 combined = (combined << B) | b 

166 

167 # Laengenfeld aus den obersten 16 Bits extrahieren. 

168 # Struktur: [16 Bit Laenge | msg_len*8 Datenbits | pad Fuellbits] 

169 # total_bits = 16 + msg_len*8 + pad, 0 <= pad < B 

170 header_shift = total_bits - 16 

171 candidate_len = (combined >> header_shift) & 0xFFFF 

172 

173 data_bits = candidate_len * 8 

174 used_bits = 16 + data_bits 

175 pad = total_bits - used_bits 

176 

177 if pad < 0 or pad >= B: 

178 raise ValueError("Ungueltiges Padding: 0x80-Byte nicht gefunden.") 

179 

180 if candidate_len == 0: 

181 return b"" 

182 

183 msg_int = (combined >> pad) & ((1 << data_bits) - 1) 

184 return msg_int.to_bytes(candidate_len, byteorder="big")

187def encrypt( 

188 pk: PublicKey, 

189 s: int, 

190 message: bytes, 

191) -> List[int]: 

192 """Verschluesselt eine Nachricht mit der Fenster-Chiffre (Algorithmus 1). 

193 

194 Parameters 

195 ---------- 

196 pk : PublicKey Oeffentlicher Schluessel (n, e, k). 

197 s : int Geheimer Offset (0 <= s < k). 

198 message : bytes Klartextnachricht (max 65535 Bytes). 

199 

200 Returns 

201 ------- 

202 List[int] Paketfolge P = (P_1, ..., P_l). 

203 Dummy-Pakete: gleichverteilte Zufallszahlen in [0, n). 

204 Echte Pakete: C_j = M_j^e mod n. 

205 """ 

206 if not (0 <= s < pk.k): 

207 raise ValueError(f"Offset s={s} liegt nicht in [0, k-1={pk.k - 1}].") 

208 

209 n, e, k = pk.n, pk.e, pk.k 

210 

211 # Schritt 1: Blockzerlegung 

212 real_blocks = message_to_blocks(message, n) 

213 r = len(real_blocks) 

214 

215 # Schritt 2: Gesamtlaenge (offset-korrekt: genau r echte Slots) 

216 total = _compute_packet_count(r, k, s) 

217 

218 # Schritt 3: Paketfolge aufbauen 

219 packets: List[int] = [] 

220 j = 0 

221 for i in range(1, total + 1): 

222 if (i - 1 + s) % k == 0: 

223 packets.append(_dummy_int(n)) 

224 else: 

225 packets.append(pow(real_blocks[j], e, n)) 

226 j += 1 

227 

228 assert j == r, f"Interner Fehler: {j} Bloecke platziert, erwartet {r}" 

229 return packets
