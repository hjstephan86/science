1""" 

2tmm.py 

3================= 

4Transfermatrix-Methode (TMM) für beliebige optische Schichtsysteme. 

5 

6Implementiert die charakteristische Matrix-Methode nach Born & Wolf (1999) 

7für s- und p-Polarisation sowie unpolarisiertes Licht. 

8 

9Gleichungen: 

10 M_j = [[cos δ_j, -i/η_j · sin δ_j], 

11 [-i η_j · sin δ_j, cos δ_j]] 

12 

13 δ_j = 2π ñ_j d_j cos θ_j / λ 

14 η_j^(s) = ñ_j cos θ_j / Z_0 

15 η_j^(p) = ñ_j / cos θ_j / Z_0 (für p-Pol.) 

16 

17Wichtige Analogie: 

18 Die TMM ist formal identisch mit dem QM-Tunnelmatrix-Formalismus. 

19"""

32class OpticalLayer:

34 

35 Parameters

38 Komplexer Brechungsindex ñ = n + iκ. 

39 d : float 

40 Schichtdicke [m]. 0 für halb-unendliches Medium.

42 Beschriftung.

44 n: complex 

45 d: float 

46 label: str = ""

53class TransferMatrixMethod: 

54 """Transfermatrix-Methode für geschichtete optische Medien. 

55 

56 Parameters 

57 ---------- 

58 incident_medium : OpticalLayer 

59 Eingangsmedium (halb-unendlich, d=0). 

60 layers : list[OpticalLayer] 

61 Schichtstapel in Ausbreitungsrichtung. 

62 substrate : OpticalLayer 

63 Substrat (halb-unendlich, d=0). 

64 theta_i_deg : float 

65 Einfallswinkel [Grad]. 

66 """ 

67 

68 def __init__( 

69 self, 

70 incident_medium: OpticalLayer, 

71 layers: List[OpticalLayer], 

72 substrate: OpticalLayer, 

73 theta_i_deg: float = 0.0, 

74 ) -> None: 

75 self.medium = incident_medium 

76 self.layers = layers 

77 self.substrate = substrate 

78 self.theta_i = np.radians(theta_i_deg) 

79 

80 # ── Snellius ─────────────────────────────────────────────────────────────── 

81 

82 def _snell(self, n_in: complex, n_out: complex, theta_in: complex) -> complex: 

83 """Snellsches Brechungsgesetz: n_in sin θ_in = n_out sin θ_out.""" 

84 sin_out = n_in * np.sin(theta_in) / n_out 

85 # Für absorbierende Medien: arcsin komplex 

86 return np.arcsin(sin_out) 

87 

88 def _refraction_angles(self) -> List[complex]: 

89 """Brechungswinkel in allen Schichten.""" 

90 all_n = [self.medium.n] + [l.n for l in self.layers] + [self.substrate.n] 

91 angles = [self.theta_i] 

92 for i in range(1, len(all_n)): 

93 theta_next = self._snell(all_n[i - 1], all_n[i], angles[-1]) 

94 angles.append(theta_next) 

95 return angles 

96 

97 # ── Admittanzen ──────────────────────────────────────────────────────────── 

98 

99 def _admittance_s(self, n: complex, theta: complex) -> complex: 

100 """Optische Admittanz η_s = n cos θ / Z_0.""" 

101 return n * np.cos(theta) / Z_0 

102 

103 def _admittance_p(self, n: complex, theta: complex) -> complex: 

104 """Optische Admittanz η_p = n / (cos θ · Z_0).""" 

105 cos_t = np.cos(theta) 

106 if np.abs(cos_t) < 1e-12: 

107 return np.inf 

108 return n / (cos_t * Z_0) 

109 

110 # ── Charakteristische Matrix ─────────────────────────────────────────────── 

111 

112 def _layer_matrix_s( 

113 self, 

114 n: complex, 

115 d: float, 

116 theta: complex, 

117 wavelength: float, 

118 ) -> np.ndarray: 

119 """Charakteristische Matrix M_j für s-Polarisation. 

120 

121 M_j = [[cos δ, -i/η sin δ], 

122 [-iη sin δ, cos δ ]] 

123 """ 

124 delta = 2.0 * np.pi * n * d * np.cos(theta) / wavelength 

125 eta = self._admittance_s(n, theta) 

126 cos_d = np.cos(delta) 

127 sin_d = np.sin(delta) 

128 return np.array([ 

129 [cos_d, -1j / eta * sin_d], 

130 [-1j * eta * sin_d, cos_d ], 

131 ]) 

132 

133 def _layer_matrix_p( 

134 self, 

135 n: complex, 

136 d: float, 

137 theta: complex, 

138 wavelength: float, 

139 ) -> np.ndarray: 

140 """Charakteristische Matrix M_j für p-Polarisation.""" 

141 delta = 2.0 * np.pi * n * d * np.cos(theta) / wavelength 

142 eta = self._admittance_p(n, theta) 

143 cos_d = np.cos(delta) 

144 sin_d = np.sin(delta) 

145 return np.array([ 

146 [cos_d, -1j / eta * sin_d], 

147 [-1j * eta * sin_d, cos_d ], 

148 ]) 

149 

150 # ── Gesamtmatrix ─────────────────────────────────────────────────────────── 

151 

152 def _system_matrix( 

153 self, 

154 wavelength: float, 

155 polarization: str = "s", 

156 ) -> np.ndarray: 

157 """Gesamtmatrix M_ges = ∏ M_j.""" 

158 angles = self._refraction_angles() 

159 mat_fn = self._layer_matrix_s if polarization == "s" else self._layer_matrix_p 

160 M = np.eye(2, dtype=complex) 

161 for i, layer in enumerate(self.layers): 

162 M_j = mat_fn(layer.n, layer.d, angles[i + 1], wavelength) 

163 M = M @ M_j 

164 return M 

165 

166 # ── Reflexions- und Transmissionskoeffizienten ───────────────────────────── 

167 

168 def _rt_coefficients( 

169 self, 

170 wavelength: float, 

171 polarization: str = "s", 

172 ) -> Tuple[complex, complex]: 

173 """Reflexions- und Transmissionsamplitude (r, t). 

174 

175 Aus Gleichung (7.3.x) Born & Wolf. 

176 """ 

177 angles = self._refraction_angles() 

178 adm_fn = self._admittance_s if polarization == "s" else self._admittance_p 

179 eta_0 = adm_fn(self.medium.n, angles[0]) 

180 eta_s = adm_fn(self.substrate.n, angles[-1]) 

181 M = self._system_matrix(wavelength, polarization) 

182 m11, m12, m21, m22 = M[0, 0], M[0, 1], M[1, 0], M[1, 1] 

183 numerator_r = m11 * eta_0 + m12 * eta_0 * eta_s - m21 - m22 * eta_s 

184 denominator = m11 * eta_0 + m12 * eta_0 * eta_s + m21 + m22 * eta_s 

185 if np.abs(denominator) < 1e-30: 

186 return 1.0 + 0j, 0.0 + 0j 

187 r = numerator_r / denominator 

188 t = 2.0 * eta_0 / denominator 

189 return r, t

191 # ── Öffentliche API ──────────────────────────────────────────────────────── 

192 

193 def reflectance( 

194 self, 

195 wavelength: float, 

196 polarization: str = "avg", 

197 ) -> float: 

198 """Reflexionsgrad R = |r|² bei gegebener Wellenlänge. 

199 

200 Parameters 

201 ---------- 

202 wavelength : float 

203 Wellenlänge [m]. 

204 polarization : str 

205 's', 'p', oder 'avg' (Mittelwert). 

206 

207 Returns 

208 ------- 

209 float 

210 Reflexionsgrad R ∈ [0, 1]. 

211 """ 

212 if polarization == "avg": 

213 rs, _ = self._rt_coefficients(wavelength, "s") 

214 rp, _ = self._rt_coefficients(wavelength, "p") 

215 return 0.5 * (abs(rs)**2 + abs(rp)**2) 

216 else: 

217 r, _ = self._rt_coefficients(wavelength, polarization) 

218 return abs(r) ** 2 

219 

220 def transmittance( 

221 self, 

222 wavelength: float, 

223 polarization: str = "avg", 

224 ) -> float: 

225 """Transmissionsgrad T = 1 − R (für verlustfreie Medien).""" 

226 return 1.0 - self.reflectance(wavelength, polarization)

228 def spectrum( 

229 self, 

230 lam_min_nm: float = 380.0, 

231 lam_max_nm: float = 780.0, 

232 n_points: int = 400, 

233 polarization: str = "avg", 

234 ) -> Tuple[np.ndarray, np.ndarray]: 

235 """Berechnetes Reflexionsspektrum R(λ). 

236 

237 Returns 

238 ------- 

239 (wavelengths_nm, R) : tuple[ndarray, ndarray] 

240 """ 

241 lam_nm = np.linspace(lam_min_nm, lam_max_nm, n_points) 

242 R = np.array([ 

243 self.reflectance(l * 1e-9, polarization) for l in lam_nm 

244 ]) 

245 return lam_nm, R

247 def phase_matrix(self, wavelength: float) -> np.ndarray: 

248 """Rückgabe der Gesamtphasenmatrix (für QM-Analogie). 

249 

250 Die formale Identität mit dem QM-Zeitentwicklungsoperator 

251 wird durch diese Struktur sichtbar. 

252 """ 

253 return self._system_matrix(wavelength, "s")

255 def ellipsometry_params(self, wavelength: float) -> Tuple[float, float]: 

256 """Ellipsometrie-Parameter Ψ und Δ. 

257 

258 tan Ψ · e^(iΔ) = r_p / r_s 

259 

260 Returns 

261 ------- 

262 (Psi_deg, Delta_deg) : tuple[float, float] 

263 """ 

264 rp, _ = self._rt_coefficients(wavelength, "p") 

265 rs, _ = self._rt_coefficients(wavelength, "s") 

266 if abs(rs) < 1e-30: 

267 return 0.0, 0.0 

268 rho = rp / rs 

269 psi = np.degrees(np.arctan(abs(rho))) 

270 delta = np.degrees(np.angle(rho)) 

271 return psi, delta

282def fresnel_rs(n1: complex, n2: complex, theta_i_deg: float = 0.0) -> complex: 

283 """Fresnel-Amplitudenreflexionskoeffizient r_s an ebener Grenzfläche.""" 

284 theta_i = np.radians(theta_i_deg) 

285 sin_t = np.sin(theta_i) 

286 cos_i = np.cos(theta_i) 

287 cos_t = np.sqrt(1.0 - (n1 / n2 * sin_t)**2 + 0j) 

288 return (n1 * cos_i - n2 * cos_t) / (n1 * cos_i + n2 * cos_t) 

289 

290 

291def fresnel_rp(n1: complex, n2: complex, theta_i_deg: float = 0.0) -> complex: 

292 """Fresnel-Amplitudenreflexionskoeffizient r_p an ebener Grenzfläche.""" 

293 theta_i = np.radians(theta_i_deg) 

294 sin_t = np.sin(theta_i) 

295 cos_i = np.cos(theta_i) 

296 cos_t = np.sqrt(1.0 - (n1 / n2 * sin_t)**2 + 0j) 

297 return (n2 * cos_i - n1 * cos_t) / (n2 * cos_i + n1 * cos_t)

300def build_hydrogel_stack( 

301 d0_nm: float = 250.0, 

302 n_polymer: float = 1.40, 

303 Q: float = 1.0, 

304 R_mirror: float = 0.5, 

305) -> TransferMatrixMethod: 

306 """Erstellt einen TMM-Stack für ein Hydrogel-FP-System. 

307 

308 Aufbau: Luft | Au-Spiegel | Hydrogel | Au-Spiegel | Glas 

309 

310 Parameters 

311 ---------- 

312 d0_nm : float 

313 Trockene Hydrogeldicke [nm]. 

314 n_polymer : float 

315 Brechungsindex des trockenen Polymers. 

316 Q : float 

317 Quellungsgrad. 

318 R_mirror : float 

319 Spiegel-Reflexivität. 

320 

321 Returns 

322 ------- 

323 TransferMatrixMethod 

324 """ 

325 n_gel = 1.0 + (n_polymer - 1.0) / Q 

326 d_gel = Q * d0_nm * 1e-9 

327 d_mirror = 10e-9 # 10 nm Au-Spiegel 

328 n_au = 0.47 + 2.83j # Gold bei 550 nm (Näherung) 

329 layers = [ 

330 OpticalLayer(n_au, d_mirror, "Au-Spiegel 1"), 

331 OpticalLayer(n_gel, d_gel, f"Hydrogel (Q={Q:.1f})"), 

332 OpticalLayer(n_au, d_mirror, "Au-Spiegel 2"), 

333 ] 

334 medium = OpticalLayer(1.0, 0.0, "Luft") 

335 substrate = OpticalLayer(1.52, 0.0, "Borosilikatglas") 

336 return TransferMatrixMethod(medium, layers, substrate)
