213def shannon_capacity(B: float, snr_db: float) -> float:
214    """Shannon channel capacity C = B · log₂(1 + S/N) [bit/s].
215
216    Parameters
217    ----------
218    B:
219        Bandwidth [Hz].  Must be positive.
220    snr_db:
221        Signal-to-noise ratio [dB].
222
223    Raises
224    ------
225    ValueError
226        If *B* ≤ 0.
227    """
228    if B <= 0:
229        raise ValueError(f"Bandwidth must be positive, got B={B}")
230    snr_linear = 10.0 ** (snr_db / 10.0)
231    return float(B * np.log2(1.0 + snr_linear))

234def membrane_efficiency_omega(
235    profile: HearingProfile,
236    reference: HearingProfile = CAT,
237) -> float:
238    """Composite membrane efficiency criterion Ω.
239
240    .. math::
241        \\Omega = \\frac{\\eta_B}{\\eta_{B,\\text{ref}}}
242                  \\cdot \\frac{\\bar{Q}}{\\bar{Q}_{\\text{ref}}}
243                  \\cdot \\frac{D}{D_{\\text{ref}}}
244                  \\cdot \\frac{1/\\sigma_t}{1/\\sigma_{t,\\text{ref}}}
245
246    All four terms are normalised to the reference profile (default: cat).
247
248    Parameters
249    ----------
250    profile:
251        Hearing profile of the species to evaluate.
252    reference:
253        Reference profile (Ω = 1 for the reference).
254
255    Returns
256    -------
257    float
258        Dimensionless Ω score.
259    """
260    eta_ratio = profile.bandwidth_efficiency / reference.bandwidth_efficiency
261    q_ratio = profile.Q_mean / reference.Q_mean
262    d_ratio = profile.dynamic_range_db / reference.dynamic_range_db
263    t_ratio = reference.sigma_t / profile.sigma_t   # higher temporal res → larger ratio
264    return float(eta_ratio * q_ratio * d_ratio * t_ratio)

267def compare_profiles(*profiles: HearingProfile) -> Dict[str, dict]:
268    """Return a structured comparison dictionary for one or more profiles.
269
270    Parameters
271    ----------
272    *profiles:
273        One or more :class:`HearingProfile` instances.
274
275    Returns
276    -------
277    dict
278        Keyed by ``profile.species``, containing all relevant metrics.
279    """
280    result: Dict[str, dict] = {}
281    ref = CAT
282    for p in profiles:
283        result[p.species] = {
284            "latin": p.latin_name,
285            "f_min_Hz": p.f_min,
286            "f_max_Hz": p.f_max,
287            "bandwidth_Hz": p.bandwidth,
288            "L_membrane_mm": p.L_membrane,
289            "beta_per_mm": p.beta,
290            "Q_mean": p.Q_mean,
291            "Q_range": p.Q_range,
292            "sigma_t_us": p.sigma_t * 1e6,
293            "dynamic_range_dB": p.dynamic_range_db,
294            "bandwidth_efficiency_kHz_per_mm": p.bandwidth_efficiency / 1e3,
295            "temporal_resolution_kHz": p.temporal_resolution_hz / 1e3,
296            "shannon_capacity_Mbit_per_s": p.shannon_capacity_bits_per_second / 1e6,
297            "omega": membrane_efficiency_omega(p, reference=ref),
298        }
299    return result
