uqtl

uqtl – Unified Query Translation Layer

Version 2.0 des UQTL-Standards für die sprachunabhängige Darstellung und Äquivalenzprüfung relationaler Datenbankabfragen.

Paketstruktur

uqtl.iqt Alle Datenklassen des Intermediate Query Tree (IQT): Knotentypen, Ausdruckstypen, Enumerationen und Hilfsklassen.

uqtl.normalizer ~uqtl.normalizer.IQTNormalizer – setzt die fünf Normalisierungsregeln (3.1–3.5) um und bringt jeden IQT-Baum in kanonische Form.

uqtl.sql_generator ~uqtl.sql_generator.SQLGenerator – wandelt einen normalisierten IQT-Baum in ANSI SQL:2016 um.

uqtl.equivalence ~uqtl.equivalence.IQTEquivalenceChecker – zweistufige Äquivalenzprüfung via Subgraph Algorithmus und SQL-Vergleich.

Schnellstart

::

from uqtl import (
    ScanNode, FilterNode, ProjectNode,
    NodeType, CompareOp, CompareExpr, ColumnRef, Literal,
    ProjectionItem,
    IQTNormalizer, SQLGenerator, IQTEquivalenceChecker,
)

scan = ScanNode(node_type=NodeType.SCAN, table="orders", alias="o")
filt = FilterNode(
    node_type=NodeType.FILTER,
    source=scan,
    predicate=CompareExpr(
        ColumnRef("status", "o"), CompareOp.EQ, Literal("delivered")
    ),
)

sql = SQLGenerator().generate(IQTNormalizer().normalize(filt))
print(sql)
# SELECT *
# FROM orders AS t1
# WHERE t1.status = 'delivered'

checker = IQTEquivalenceChecker()
print(checker.are_equivalent(filt, filt))  # True

419    def _simplify_boolean(self, expr: Expr) -> Expr:
420        """Eliminiert redundante boolesche Ausdrücke (Regel 3.3).
421
422        Transformationen:
423          - ``p AND TRUE``  → ``p``
424          - ``p OR FALSE``  → ``p``
425          - ``NOT NOT p``   → ``p``
426          - ``p AND p``     → ``p``  (Deduplizierung)
427          - ``p AND [TRUE, TRUE]`` → alle TRUE-Literale werden entfernt
428
429        Args:
430            expr: Zu vereinfachender Ausdruck.
431
432        Returns:
433            Vereinfachter Ausdruck (ggf. unverandert, falls keine Regel greift).
434        """
435        if not isinstance(expr, LogicExpr):
436            return expr
437
438        if expr.op == LogicOp.NOT:
439            inner = expr.operands[0] if expr.operands else expr
440            # NOT NOT p → p
441            if isinstance(inner, LogicExpr) and inner.op == LogicOp.NOT:
442                return self._simplify_boolean(inner.operands[0])
443            return LogicExpr(op=LogicOp.NOT, operands=[self._simplify_boolean(inner)])
444
445        simplified = [self._simplify_boolean(o) for o in expr.operands]
446
447        if expr.op == LogicOp.AND:
448            # Entferne TRUE-Literale
449            filtered = [
450                o for o in simplified
451                if not (isinstance(o, Literal) and o.value is True)
452            ]
453            # Deduplizierung
454            seen: List[str] = []
455            deduped = []
456            for o in filtered:
457                key = repr(o)
458                if key not in seen:
459                    seen.append(key)
460                    deduped.append(o)
461            if len(deduped) == 1:
462                return deduped[0]
463            return LogicExpr(op=LogicOp.AND, operands=deduped)
464
465        if expr.op == LogicOp.OR:
466            # Entferne FALSE-Literale
467            filtered = [
468                o for o in simplified
469                if not (isinstance(o, Literal) and o.value is False)
470            ]
471            if len(filtered) == 1:
472                return filtered[0]
473            return LogicExpr(op=LogicOp.OR, operands=filtered)
474
475        return LogicExpr(op=expr.op, operands=simplified)
476
477    # ── AND-Prädikat-Sortierung (Regel 3.1) ──────────────────────────────────

479    @staticmethod
480    def _sort_and_predicates(expr: LogicExpr) -> LogicExpr:
481        """Sortiert AND-Operanden lexikografisch nach ``repr()`` (Regel 3.1).
482
483        Damit gilt nach der Normalisierung:
484        ``p1 AND p2 ≡ p2 AND p1``
485        (beide erzeugen dieselbe normalisierte Darstellung).
486
487        Args:
488            expr: Ein ``LogicExpr`` mit ``op == LogicOp.AND``.
489
490        Returns:
491            Ein neues ``LogicExpr`` mit sortierten Operanden.
492        """
493        sorted_ops = sorted(expr.operands, key=repr)
494        return LogicExpr(op=LogicOp.AND, operands=sorted_ops)

421    def _gen_expr(self, expr: Expr) -> str:
422        """Wandelt einen Ausdrucksknoten rekursiv in SQL um.
423
424        Unterstützte Ausdruckstypen:
425
426        * :class:`~uqtl.iqt.ColumnRef` → ``table.column`` oder ``column``
427        * :class:`~uqtl.iqt.Literal` → ``NULL``, ``TRUE``/``FALSE``, ``'str'``, Zahl
428        * :class:`~uqtl.iqt.FuncCall` → ``FUNC([DISTINCT] args)``
429        * :class:`~uqtl.iqt.CompareExpr` → ``left op right``
430        * :class:`~uqtl.iqt.LogicExpr` → ``a AND b``, ``a OR b``, ``NOT (a)``
431        * :class:`~uqtl.iqt.InExpr` → ``expr [NOT] IN (items|subquery)``
432        * :class:`~uqtl.iqt.BetweenExpr` → ``expr [NOT] BETWEEN lo AND hi``
433        * :class:`~uqtl.iqt.LikeExpr` → ``expr [NOT] LIKE 'pattern' [ESCAPE 'c']``
434        * :class:`~uqtl.iqt.NullExpr` → ``expr IS [NOT] NULL``
435        * :class:`~uqtl.iqt.ExistsExpr` → ``[NOT] EXISTS (subquery)``
436
437        Args:
438            expr: Zu übersetzender Ausdrucksknoten.
439
440        Returns:
441            SQL-Teilstring des Ausdrucks.
442        """
443        if isinstance(expr, ColumnRef):
444            if expr.table:
445                return f"{expr.table}.{expr.column}"
446            return expr.column
447
448        if isinstance(expr, Literal):
449            if expr.value is None:
450                return "NULL"
451            if isinstance(expr.value, bool):
452                return "TRUE" if expr.value else "FALSE"
453            if isinstance(expr.value, str):
454                escaped = expr.value.replace("'", "''")
455                return f"'{escaped}'"
456            return str(expr.value)
457
458        if isinstance(expr, FuncCall):
459            distinct_kw = "DISTINCT " if expr.distinct else ""
460            args = ", ".join(self._gen_expr(a) for a in expr.args)
461            return f"{expr.name.upper()}({distinct_kw}{args})"
462
463        if isinstance(expr, CompareExpr):
464            left = self._gen_expr(expr.left)
465            right = self._gen_expr(expr.right)
466            return f"{left} {expr.op.value} {right}"
467
468        if isinstance(expr, LogicExpr):
469            if expr.op == LogicOp.NOT:
470                inner = self._gen_expr(expr.operands[0])
471                return f"NOT ({inner})"
472            sep = f" {expr.op.value} "
473            parts = [self._gen_expr(o) for o in expr.operands]
474            return sep.join(parts)
475
476        if isinstance(expr, InExpr):
477            val = self._gen_expr(expr.value)
478            not_kw = "NOT " if expr.negated else ""
479            if expr.subquery:
480                inner = self._gen_node(expr.subquery.query)
481                return f"{val} {not_kw}IN ({inner})"
482            items = ", ".join(self._gen_expr(i) for i in expr.items)
483            return f"{val} {not_kw}IN ({items})"
484
485        if isinstance(expr, BetweenExpr):
486            val = self._gen_expr(expr.value)
487            lo = self._gen_expr(expr.lo)
488            hi = self._gen_expr(expr.hi)
489            not_kw = "NOT " if expr.negated else ""
490            return f"{val} {not_kw}BETWEEN {lo} AND {hi}"
491
492        if isinstance(expr, LikeExpr):
493            val = self._gen_expr(expr.value)
494            not_kw = "NOT " if expr.negated else ""
495            escape = f" ESCAPE '{expr.escape}'" if expr.escape else ""
496            return f"{val} {not_kw}LIKE '{expr.pattern}'{escape}"
497
498        if isinstance(expr, NullExpr):
499            val = self._gen_expr(expr.value)
500            return f"{val} IS NOT NULL" if expr.negated else f"{val} IS NULL"
501
502        if isinstance(expr, ExistsExpr):
503            inner = self._gen_node(expr.subquery.query)
504            not_kw = "NOT " if expr.negated else ""
505            return f"{not_kw}EXISTS (\n    {inner}\n)"
506
507        return str(expr)

188    def _collect_nodes(self, root: IQTNode) -> List[IQTNode]:
189        """Sammelt alle Knoten des IQT-Baums in Pre-Order-Reihenfolge.

213        bei binären Knoten (JoinNode, SetOpNode) zuerst links, dann rechts.

248    def _build_adjacency_matrix(self, root: IQTNode) -> np.ndarray:
249        """
250        Kodiert den IQT-Baum als quadratische Adjazenzmatrix.
251
252        Knoten werden als Zeilen/Spalten kodiert (Pre-Order-Index).
253        Eine 1 an Position [i, j] bedeutet: Knoten i ist Elternteil von j.
254
255        Der Knotentyp wird als Diagonaleintrag gespeichert (1 = belegt),
256        um die Knotenanzahl für den Subgraph Algorithmus zu erhalten.
257        """
258        nodes = self._collect_nodes(root)
259        n = len(nodes)
260        idx = {id(node): i for i, node in enumerate(nodes)}
261
262        matrix = np.zeros((n, n), dtype=int)
263
264        for node in nodes:
265            i = idx[id(node)]
266            # Diagonale markiert vorhandenen Knoten
267            matrix[i, i] = 1
268            for child in self._children_of(node):
269                j = idx.get(id(child))
270                if j is not None:
