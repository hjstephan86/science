liionp

liionp – KI-Power-Management-Modul für Lithium-Ionen-Akkumulatoren.

Dieses Paket stellt alle Bausteine bereit, die in liionp.tex beschrieben werden: Degradationsmodelle, thermische Modelle, Zellmodelle, Regler, Mechanikmodelle, Analyseformeln sowie einen Simulationsmotor zum Vergleich von statischem PMS und KI-PMM.

44    """Relative Ratenbeschleunigung von *T0* nach *T* (Gl. 6).

64    """Momentane SEI-Schichtdicken-Wachstumsrate (Gl. 3).

76    """

125def run_simulation(
126    profile: LoadProfile,
127    params: CellParameters | None = None,
128    ai_mode: bool = False,
129    n_cycles: int = 20,
130    dt: float = 60.0,
131    T0: float = 298.0,
132) -> SimulationResult:
133    """Führt eine vollständige Zyklussimulation durch.
134
135    Die beiden Betriebsmodi unterscheiden sich wie folgt:
136
137    * **Statisches PMS** (``ai_mode=False``): festes Temperaturziel 308 K;
138      keine Stromreduzierung zur Vorkühlung.
139    * **KI-PMM** (``ai_mode=True``): niedrigeres Temperaturziel 303 K;
140      Strom wird reduziert, wenn die Zelle den thermischen Soft-Limit-Wert erreicht.
141
142    Args:
143        profile:  Lastprofil, das I(t) und T_amb bereitstellt.
144        params:   Zellparameter (Standard-NMC-Werte wenn *None*).
145        ai_mode:  KI-PMM-Betrieb aktivieren.
146        n_cycles: Maximale Anzahl zu simulierender Zyklen.
147        dt:       Simulationszeitschritt [s].
148        T0:       Anfangszelltemperatur [K].
149
150    Returns:
151        :class:`SimulationResult` mit vollständigen Zeitreihendaten.
152    """
153    if params is None:
154        params = CellParameters()
155
156    model = CellModel(params)
157    state = CellState.initial(params, T0)
158
159    T_soft_limit = 303.0 if ai_mode else 308.0  # K  soft thermal ceiling
160
161    # Pre-allocate lists
162    times_out: list[float] = []
163    temps_out: list[float] = []
164    volts_out: list[float] = []
165    socs_out: list[float] = []
166    degs_out: list[float] = []
167    rates_out: list[float] = []
168
169    t = 0.0
170    t_cycle = 3600.0  # treat 1 h as one "cycle" unit
171    cycles_done = 0
172
173    while cycles_done < n_cycles and not state.is_eol:
174        I = profile.current_at(t)
175
176        # AI-PMM: predictive thermal management — reduce current near the limit
177        if ai_mode and state.T > T_soft_limit:
178            I = I * 0.7
179
180        new_state, deg_rate = model.step(state, I, profile.T_amb, dt)
181        state = new_state
182
183        times_out.append(t)
184        temps_out.append(state.T)
185        volts_out.append(state.V)
186        socs_out.append(state.SoC)
187        degs_out.append(state.degradation)
188        rates_out.append(deg_rate)
189
190        t += dt
191        # Count completed cycles (each t_cycle seconds = 1 cycle)
192        if (t % t_cycle) < dt:
193            cycles_done += 1
194
195    times_arr = np.array(times_out)
196    temps_arr = np.array(temps_out)
197    rates_arr = np.array(rates_out)
198
199    return SimulationResult(
200        name="AI-PMM" if ai_mode else "Static PMS",
201        times=times_arr,
202        temperatures=temps_arr,
203        voltages=np.array(volts_out),
204        SoCs=np.array(socs_out),
205        degradations=np.array(degs_out),
206        degradation_rates=rates_arr,
207        cycles_completed=cycles_done,
208        mean_temperature=float(np.mean(temps_arr)) if len(temps_arr) else T0,
209        mean_degradation_rate=float(np.mean(rates_arr)) if len(rates_arr) else 0.0,
210    )
