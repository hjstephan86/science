192    def _is_subgraph(self, i: int, j: int) -> bool:
193        """
194        Oracle call: True iff G_i is subgraph-isomorphic to G_j.
195
196        Uses Subgraph.compare_graphs(A_i, A_j) and maps the returned
197        decision string to a boolean.
198        """
199        decision, _ = self._oracle.compare_graphs(self._graphs[i], self._graphs[j])
200        return decision in _I_SUBGRAPH_OF_J
201
202    def _build_meta_graph(self) -> MetaGraph:
203        """
204        Phase 1 – Meta-Graph Construction.
205
206        Iterates over all ordered pairs (i, j) with i ≠ j and invokes the
207        Subgraph Algorithm oracle.  For every pair where G_i ⊆ G_j the
208        directed edge i → j is inserted into the meta-graph.
209
210        Time complexity: Θ(N² · T_S(n)) = O(n⁵) with T_S = O(n³), N = n.
211        """
212        n = len(self._graphs)
213        meta = MetaGraph(n=n)
214        for i in range(n):
215            for j in range(n):
216                if i != j and self._is_subgraph(i, j):
217                    meta.add_edge(i, j)
218        return meta

221    # Phase 2: DFS Forest Computation
222    # ------------------------------------------------------------------

224    @staticmethod
225    def _dfs_forest(meta: MetaGraph) -> DFSResult:
226        """
227        Phase 2 – DFS Forest Computation.
228
229        Runs standard DFS on the meta-graph, assigning discovery/finish
230        timestamps and classifying every edge as TREE, BACK, FORWARD, or
231        CROSS.
232
233        Time complexity: Θ(N + |E|) = O(n²) in the worst case.
234        """
235        _WHITE, _GRAY, _BLACK = 0, 1, 2
236
237        color: Dict[int, int] = {i: _WHITE for i in range(meta.n)}
238        parent: Dict[int, Optional[int]] = {i: None for i in range(meta.n)}
239        d: Dict[int, int] = {}
240        f: Dict[int, int] = {}
241        chi: Dict[Tuple[int, int], EdgeClass] = {}
242        time = [0]
243        roots: List[int] = []
244
245        def _classify(u: int, v: int) -> EdgeClass:
246            if color[v] == _WHITE:
247                return EdgeClass.TREE
248            if color[v] == _GRAY:
249                return EdgeClass.BACK
250            return EdgeClass.FORWARD if d[u] < d[v] else EdgeClass.CROSS
251
252        def _visit(u: int) -> None:
253            color[u] = _GRAY
254            time[0] += 1
255            d[u] = time[0]
256            for v in meta.neighbors(u):
257                chi[(u, v)] = _classify(u, v)
258                if color[v] == _WHITE:
259                    parent[v] = u
260                    _visit(v)
261            color[u] = _BLACK
262            time[0] += 1
263            f[u] = time[0]
264
265        for i in range(meta.n):
266            if color[i] == _WHITE:
267                roots.append(i)
268                _visit(i)
269
270        return DFSResult(
271            discovery=d,
272            finish=f,
273            parent=parent,
274            edge_class=chi,
275            forest_roots=roots,
276        )
